<script type="text/javascript">
  function initialize() {};
  window.showRegistration = function() {
    var $box   = document.getElementById('box')  ,
        $modal = document.getElementById('modal'),
        $template = document.getElementById('registrationForm');

    $box.style.removeProperty('display');
    $modal.innerHTML = $template.innerHTML;
    fadeModal();
  }

  window.hideRegistration = function() {
    var $box = document.getElementById('box'),
        $modal = document.getElementById('modal');
    $box.style.setProperty('display', 'none');
    $box.style.setProperty('opacity', '0');
    $modal.innerHTML = '';
  }

  window.fadeModal = function() {
    var $box = document.getElementById('box');
    var opacity = parseFloat($box.style.getPropertyValue('opacity')) + 0.1;
    $box.style.setProperty('opacity', '' + opacity);

    if(opacity < 1.0) {
      setTimeout(fadeModal, 25);
    }
  };
</script>
